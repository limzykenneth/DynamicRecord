        const result = [];

    /**
     * Save all the model instances in the DynamicCollection.
     *
     * Simply calls all the individual model's `save()` method.
     *
     * @method saveAll
     * @memberOf DynamicCollection
     * @instance
     * @return {Promise}
     */
    saveAll() {
        const promises = [];

            if (model.save) {

                promises.push(model.save());

        });
